import java.util.Scanner; public class ConversaoCelsiusFahrenheit { public static void main(String[] args) { Scanner entrada = new Scanner(System.in); // Entrada de dados System.out.print("Digite a temperatura em graus Celsius: "); double celsius = entrada.nextDouble(); // Cálculo da conversão double fahrenheit = (celsius * 9 / 5) + 32; // Saída do resultado System.out.println("A temperatura em Fahrenheit é: " + fahrenheit); entrada.close(); } }
